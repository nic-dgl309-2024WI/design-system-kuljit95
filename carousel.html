Java Script
    
const carouselInner = document.querySelector('.c-carousel-inner');
  const testimonials = document.querySelectorAll('.c-testimonial');
  const testimonialWidth = testimonials[0].offsetWidth;
  let index = 0;

  function slideTestimonials() {
    carouselInner.style.transition = 'transform 0.5s ease';
    carouselInner.style.transform = `translateX(${-testimonialWidth * index}px)`;
  }

  function nexttestimonial() {
    if (index >= testimonials.length / 2) {
      index = 0;
      carouselInner.style.transition = 'none';
      carouselInner.style.transform = `translateX(0)`;
      setTimeout(() => {
        index++;
        slideTestimonials();
      }, 50);
    } else {
      index++;
      slideTestimonials();
    }
  }

  function prevtestimonial() {
    if (index <= 0) {
      index = testimonials.length / 2;
      carouselInner.style.transition = 'none';
      carouselInner.style.transform = `translateX(-${testimonialWidth * (testimonials.length / 2)}px)`;
      setTimeout(() => {
        index--;
        slideTestimonials();
      }, 50);
    } else {
      index--;
      slideTestimonials();
    }
  }
